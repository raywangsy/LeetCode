## [234. 回文链表](https://leetcode.cn/problems/palindrome-linked-list/)

---

### 方法一：快慢指针

- 链表的反转（递归，迭代）
- 取中间节点（快慢指针）

```java
class Solution {
    public boolean isPalindrome(ListNode head) {
        if (head == null || head.next == null) {
            return true;
        }
        if (head.next.next == null) {
            return head.val == head.next.val;
        }
        // 中间节点
        ListNode midNode = midNode(head);
        // 将中间节点的右半部分翻转, rh 是右半部分翻转后的头节点
        ListNode rh = reverseList(midNode.next);
        ListNode lh = head;
        while (rh != null) {
            if (rh.val != lh.val) {
                return false;
            }
            rh = rh.next;
            lh = lh.next;
        }
        return true;
    }

    /**
     * 翻转链表
     *
     * @param head 原链表的头节点
     * @return 翻转后的链表头节点
     */
    private ListNode reverseList(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        ListNode newHead = reverseList(head.next);
        head.next.next = head;
        head.next = null;
        return newHead;
    }

    // 中间节点：即右半部分链表头节点的前一个节点
    // 比如：1>2>3>2>1 的中间节点是 3
    // 比如：1>2>2>1 的中间节点是第一个 2
    // 方法是使用快慢指针 f 一次走 2 步，s 一次走 1 步
    // 当 f.next == null 或 f.next.next == null 时，s 刚好在中间节点上
    private ListNode midNode(ListNode head) {
        ListNode fast = head;
        ListNode slow = head;
        while (fast.next != null && fast.next.next != null) {
            fast = fast.next.next;
            slow = slow.next;
        }
        return slow;
    }
}
```
